# >- INIT -<
country_event = {
	id = hga_menu.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		NOT = { has_global_flag = hga_menu_initialized }
	}

	immediate = {
		set_global_flag = hga_menu_initialized
		set_country_flag = hga_is_host

		## DEFAULT SETTING SETTER
		set_global_flag = hga_detox_tech_enabled
		set_global_flag = hga_features_intensity_medium
		set_global_flag = hga_features_production_medium

		set_global_flag = hga_arcology_menials_in_play

		# Crisis overrides >> default OFF
		# Do not set hga_crisis_type_override or hga_crisis_time_override by default

		# Relic Wars >> default ON
		# Do not set hga_no_relic_war or hga_ai_no_relic_war by default

		# Relic Trade >> default ON
		# Do not set hga_no_relic_trade or hga_ai_no_relic_trade by default

		country_event = { id = hga_menu.1 days = 1 }
	}
}

# Includes: Detox tech toggle, Crisis Type Override, Crisis Time Override
country_event = {
	id = hga_menu.10
	title = hga_menu.10.name
	desc = hga_menu.10.desc
	picture = GFX_evt_star_chart

	is_triggered_only = yes

	# <<->> Detoxification Tech <<->>

	option = {

		allow = {
			custom_tooltip = {
				OR = {
					NOT = { has_global_flag = hga_game_started }
					has_global_flag = hga_game_started_override
				}
				fail_text = hga_game_no_override_after_start
			}
		}

		trigger = {
			has_global_flag = hga_detox_tech_enabled
		}
		name = hga_menu.10.release_detox_tech
		custom_tooltip = hga_menu.10.release_detox_techTT
		hidden_effect = {
			remove_global_flag = hga_detox_tech_enabled
			country_event = { id = hga_menu.10 }
		}
	}

	option = {

		allow = {
			custom_tooltip = {
				OR = {
					NOT = { has_global_flag = hga_game_started }
					has_global_flag = hga_game_started_override
				}
				fail_text = hga_game_no_override_after_start
			}
		}

		trigger = {
			NOT = { has_global_flag = hga_detox_tech_enabled }
		}
		name = hga_menu.10.force_detox_tech
		custom_tooltip = hga_menu.100.force_detox_techTT
		hidden_effect = {
			set_global_flag = hga_detox_tech_enabled
			country_event = { id = hga_menu.10 }
		}
	}

	# <<->> Crisis Type Override (allow multiple endgame crises) <<->>

	option = {
		default_hide_option = yes

		name = hga_menu.190.crisis_type_override_on
		custom_tooltip = hga_menu.190.crisis_type_override_onTT

		trigger = {
			has_global_flag = hga_crisis_type_override
		}

		hidden_effect = {
			switch = {
				trigger = has_global_flag

				hga_crisis_type_override = {
					remove_global_flag = hga_crisis_type_override
				}
				default = {
					set_global_flag = hga_crisis_type_override
				}
			}

			country_event = { id = hga_menu.10 }
		}
	}

	option = {
		default_hide_option = yes

		name = hga_menu.190.crisis_type_override_off
		custom_tooltip = hga_menu.190.crisis_type_override_offTT

		trigger = {
			NOT = { has_global_flag = hga_crisis_type_override }
		}

		hidden_effect = {
			switch = {
				trigger = has_global_flag

				hga_crisis_type_override = {
					remove_global_flag = hga_crisis_type_override
				}
				default = {
					set_global_flag = hga_crisis_type_override
				}
			}

			country_event = { id = hga_menu.10 }
		}
	}

	# <<->> Crisis Time Override (ignore mid/endgame year gates) <<->>

	option = {
		default_hide_option = yes

		name = hga_menu.190.crisis_time_override_on
		custom_tooltip = hga_menu.190.crisis_time_override_onTT

		trigger = {
			has_global_flag = hga_crisis_time_override
		}

		hidden_effect = {
			switch = {
				trigger = has_global_flag

				hga_crisis_time_override = {
					remove_global_flag = hga_crisis_time_override
				}
				default = {
					set_global_flag = hga_crisis_time_override
				}
			}

			country_event = { id = hga_menu.10 }
		}
	}

	option = {
		default_hide_option = yes

		name = hga_menu.190.crisis_time_override_off
		custom_tooltip = hga_menu.190.crisis_time_override_offTT

		trigger = {
			NOT = { has_global_flag = hga_crisis_time_override }
		}

		hidden_effect = {
			switch = {
				trigger = has_global_flag

				hga_crisis_time_override = {
					remove_global_flag = hga_crisis_time_override
				}
				default = {
					set_global_flag = hga_crisis_time_override
				}
			}

			country_event = { id = hga_menu.10 }
		}
	}

	# <<->> Return Button <<->>

	option = {
		name = hga_menu.back
		hidden_effect = {
			country_event = { id = hga_menu.1 }
		}
	}

country_event = {
	id = hga_menu.190
	title = hga_menu.190.name
	desc = hga_menu.190.desc
	picture = GFX_evt_khan_throne_room

	is_triggered_only = yes

		name = hga_menu.back
		hidden_effect = {
			country_event = { id = hga_menu.1 }
		}
	}

	# <<->> MID-GAME: Great Khan <<->>

	option = {
		name = hga_menu.190.khan
		custom_tooltip = hga_menu.190.khan_TT

		allow = {
			AND = {
				OR = {
					mid_game_years_passed >= 0
					has_global_flag = hga_crisis_time_override
				}
			}
		}

		trigger = {
			NOT = { has_global_flag = marauder_crisis_started }
			any_country = {
				is_country_type = dormant_marauders
			}
		}

		hidden_effect = {
			country_event = { id = hga_menu.191 random = 360 }
			country_event = { id = hga_menu.190 }
		}
	}

	# <<->> MID-GAME: Voidspawn <<->>

	option = {
		name = hga_menu.190.voidworms
		custom_tooltip = hga_menu.190.voidworms_TT

		allow = {
			AND = {
				OR = {
					mid_game_years_passed >= 0
					has_global_flag = hga_crisis_time_override
				}
			}
		}

		trigger = {
			NOT = { has_global_flag = voidworms_plague_crisis_started }
			any_country = {
				is_country_type = voidworms
			}
		}

			country_event = { id = hga_menu.192 random = 360 }

			AND = {

					has_global_flag = hga_crisis_time_override

					has_global_flag = hga_crisis_type_override
					NOT = { has_global_flag = galactic_crisis_recently_fired }
				}
			}
		}

		trigger = {
			NOT = { has_global_flag = hga_scourge_summoned }
		}

			set_global_flag = hga_scourge_summoned
