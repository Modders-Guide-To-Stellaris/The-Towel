# >- COUNTRY TICKER: BEAUTY GROWING -
country_event = {
    id = hga_modifier.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        any_planet_within_border = {
            has_planet_flag = hga_beauty_growing
        }
    }

    immediate = {
        hidden_effect = {
            every_planet_within_border = {
                limit = {
                    has_planet_flag = hga_beauty_growing
                }

                if = {
                    limit = {
                        OR = {
                            is_colony = yes
                            planet_size >= 25
                        }
                    }
                    remove_planet_flag = hga_beauty_growing

                    if = { # stop grubby jezzajs from taking the unity and settling anyway
                        limit = {
                            has_deposit = d_hga_beauty_holy_unity
                        }
                        remove_deposit = d_hga_beauty_holy_unity
                    }
                }
                else = {
                    change_planet_size = 1

                    change_variable = {
                        which = hga_beauty_age
                        value = 1
                    }

                    # >- MODIFIER TIER SWAPS -

                    # Age 5: tier 1 -> tier 2 (maturing)
                    if = {
                        limit = {
                            check_variable = {
                                which = hga_beauty_age
                                value = 5
                            }
                            has_modifier = hga_beauty_hidden_within
                        }
                        remove_modifier = hga_beauty_hidden_within

                        switch = {
                            trigger = has_global_flag
                            hga_features_production_low = {
                                add_modifier = {
                                    modifier = hga_beauty_hidden_within_maturing
                                    days = -1
                                    mult = @hga_planetary_feature_mult_low
                                }
                            }
                            hga_features_production_medium = {
                                add_modifier = {
                                    modifier = hga_beauty_hidden_within_maturing
                                    days = -1
                                    mult = @hga_planetary_feature_mult_medium
                                }
                            }
                            hga_features_production_high = {
                                add_modifier = {
                                    modifier = hga_beauty_hidden_within_maturing
                                    days = -1
                                    mult = @hga_planetary_feature_mult_high
                                }
                            }
                        }
                    }

                    # Age 10: tier 2 -> tier 3 (ancient)
                    if = {
                        limit = {
                            check_variable = {
                                which = hga_beauty_age
                                value = 10
                            }
                            has_modifier = hga_beauty_hidden_within_maturing
                        }
                        remove_modifier = hga_beauty_hidden_within_maturing

                        switch = {
                            trigger = has_global_flag
                            hga_features_production_low = {
                                add_modifier = {
                                    modifier = hga_beauty_hidden_within_ancient
                                    days = -1
                                    mult = @hga_planetary_feature_mult_low
                                }
                            }
                            hga_features_production_medium = {
                                add_modifier = {
                                    modifier = hga_beauty_hidden_within_ancient
                                    days = -1
                                    mult = @hga_planetary_feature_mult_medium
                                }
                            }
                            hga_features_production_high = {
                                add_modifier = {
                                    modifier = hga_beauty_hidden_within_ancient
                                    days = -1
                                    mult = @hga_planetary_feature_mult_high
                                }
                            }
                        }
                    }
                    # /
                }
            }
        }
    }
}

            NOT = { from = { has_planet_flag = consecrated@root.owner } }

            from = { has_planet_flag = consecrated@root.owner }
        }

    }

    trigger = {
        from = { is_planet_class = pc_gas_giant }
