    show_in_diplomacy = yes
    show_notification = yes

    potential = {
        is_country_type = default
        from = { has_relic = r_unbidden_warlock }

        has_intel_level = {
            who = from
            category = economy
            level >= 2
        }
    }

    is_valid = {
        from = { has_relic = r_unbidden_warlock }
        NOT = { is_overlord_to = from }
        OR = {
            is_subject = no
            overlord = { NOT = { is_same_value = from } }
        }
    }
}
