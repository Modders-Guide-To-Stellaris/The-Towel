hga_trade_action_r_warlock = {
    fire_and_forget = yes

    potential = {
        is_country_type = default
        has_relic = r_unbidden_warlock
        NOT = { has_country_flag = hga_cannot_trade_warlock }

        from = {
            is_country_type = default
            NOT = { has_relic = r_unbidden_warlock }

            has_intel_level = {
                who = root
                category = economy
                level >= 2
            }
        }
    }

    on_traded_effect = {
        remove_relic = r_unbidden_warlock

        from = {
            add_relic = r_unbidden_warlock
        }
    }

    ai_weight = {
        weight = 1000

        # >- HARD BLOCK -<
        modifier = {
            factor = 0
            OR = {
                has_global_flag = hga_ai_no_relic_trade
                has_country_flag = hga_ai_no_relic
            }
        }

        # >- CUSTOM: Warlock is a psionic/military mobility relic -<
        # Militarists and psi-empires cling to it; materialists don't care
        modifier = {
            factor = 0.6
            has_ethic = ethic_fanatic_militarist
        }
        modifier = {
            factor = 0.8
            has_ethic = ethic_militarist
        }
        modifier = {
            factor = 1.3
            has_ethic = ethic_fanatic_materialist
        }
        modifier = {
            factor = 1.15
            has_ethic = ethic_materialist
        }

        # >- OPINION -<
        modifier = {
            factor = 1.3
            opinion_level = { who = from level = excellent }
        }
        modifier = {
            factor = 1.1
            opinion_level = { who = from level = good }
        }
        modifier = {
            factor = 0.6
            opinion_level = { who = from level = poor }
        }
        modifier = {
            factor = 0.2
            opinion_level = { who = from level = terrible }
        }

        # >- OVERLORD / SUBJECT -<
        modifier = {
            factor = 0.3
            root = { is_overlord_to = from }
        }
        modifier = {
            factor = 1.4
            from = { is_overlord_to = root }
        }

        # >- RELATIVE FLEET POWER -<
        modifier = {
            factor = 0.6
            relative_power = { who = from category = fleet value = overwhelming }
        }
        modifier = {
            factor = 0.8
            relative_power = { who = from category = fleet value = superior }
        }
        modifier = {
            factor = 1.15
            relative_power = { who = from category = fleet value = inferior }
        }
        modifier = {
            factor = 1.3
            relative_power = { who = from category = fleet value = pathetic }
        }
    }
}
