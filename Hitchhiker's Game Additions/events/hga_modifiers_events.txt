# >- COUNTRY TICKER: BEAUTY GROWING -<
country_event = {
    id = hga_modifier.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        any_owned_planet = {
            has_planet_flag = hga_beauty_growing
        }
    }

    immediate = {
        hidden_effect = {
            every_planet_within_border = {
                limit = {
                    has_planet_flag = hga_beauty_growing
                }

                if = {
                    limit = {
                        OR = {
                            is_colony = yes
                            planet_size >= 25
                        }
                    }
                    remove_planet_flag = hga_beauty_growing
                }
                else = {
                    change_planet_size = 1

                    change_variable = {
                        which = hga_beauty_age
                        value = 1
                    }

                    # >- MODIFIER TIER SWAPS -<
                    if = {
                        limit = {
                            check_variable = {
                                which = hga_beauty_age
                                value = 5
                            }
                            has_modifier = hga_beauty_hidden_within
                        }
                        remove_modifier = hga_beauty_hidden_within
                        add_modifier = { modifier = hga_beauty_hidden_within_maturing days = -1 }
                    }

                    if = {
                        limit = {
                            check_variable = {
                                which = hga_beauty_age
                                value = 10
                            }
                            has_modifier = hga_beauty_hidden_within_maturing
                        }
                        remove_modifier = hga_beauty_hidden_within_maturing
                        add_modifier = { modifier = hga_beauty_hidden_within_ancient days = -1 }
                    }
                }
            }
        }
    }
}

ship_event = {
    id = hga_modifier.10
    title = "hga_modifier.10.name"
    picture = GFX_evt_gas_giant

    desc = {
        trigger = {
            NOT = { from = { has_planet_flag = consecrated@root.owner } }
        }
        text = hga_modifier.10.desc.addonUno
    }

    desc = {
        trigger = {
            from = { has_planet_flag = consecrated@root.owner }
        }
        text = hga_modifier.10.desc.addonDos
    }

    is_triggered_only = yes

    trigger = {
        from = { is_planet_class = pc_gas_giant }
    }

    immediate = {
        hidden_effect = {
            from = {
                planet = {
                    save_event_target_as = hga_beauty_gas_giant

                    remove_planet_flag = consecrated@from

                    remove_modifier = consecrated_quality_1
                    remove_modifier = consecrated_quality_2
                    remove_modifier = consecrated_quality_3
                    remove_modifier = consecrated_quality_4

                    change_variable = {
                        which = num_consecrated_worlds
                        value = -1
                    }
                }
            }
        }
    }

    option = {
        name = hga_modifier.10.a
        ship_event = { id = hga_modifier.11 days = 30 }
    }

    option = {
        name = hga_modifier.10.b

        allow = {
            from = { has_planet_flag = consecrated@root.owner }
        }

        root.owner = {
            set_country_flag = hga_gas_giant_consecrated_approach
        }

        ship_event = { id = hga_modifier.11 days = 30 }

        hide_option_if_not_allowed = yes
    }

    option = {
        name = hga_modifier.10.c
        owner = {
            add_tech_progress = {
                tech = tech_terrestrial_sculpting
                progress = 0.25
            }
        }
    }
}
