hga_district_food_ecology = {
    inline_script = {
        script = districts/hga_ecology_district
        ICON              = district_farming
        OVERLAY_ICON      = GFX_district_food
        RESOURCE_CATEGORY = planet_districts
        JOB_SCRIPT        = "jobs/farmers_add"
        JOB_AMOUNT        = @b3_jobs
        NAME_SCRIPT       = "districts/district_triggered_name_farming"
        FLAVOR_SCRIPT     = "districts/district_triggered_flavor_desc_farming"
        PREREQUISITE      = tech_industrial_farming
        HOUSING_INLINE    = "districts/hga_arcology_housing_food"
        CONVERT_TO        = "
			district_farming
			district_rw_farming
			district_orchard_forests
			district_hive_3
		"
    }

    destroy_trigger = {
        exists = owner
        AND = {
            owner = { is_ai = yes }
            owner = { country_uses_food = no }
        }
    }
}
