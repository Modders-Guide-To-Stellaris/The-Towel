#}

# CUSTOM_AI accepts raw modifier { } blocks for per-relic trade willingness.
# Leave as "" for the shared baseline only.

# Trade action definition
	$ACTION_KEY$ = {
		fire_and_forget = yes

		potential = {
			is_country_type = default
			has_relic = $RELIC_ID$
			NOT = { has_country_flag = $FLAG_KEY$ }
			NOT = { has_global_flag = hga_no_relic_trade }

			from = {
				is_country_type = default
				NOT = { has_relic = $RELIC_ID$ }
				has_intel_level = {
					who = root
					category = economy
					level >= 2
				}
			}
		}

		# Prevent trade while a seize war is active for this relic
		possible = {
			NOT = { has_country_flag = $FLAG_KEY$ }
		}

		on_traded_effect = {
			remove_relic = $RELIC_ID$
			from = { add_relic = $RELIC_ID$ }
		}

		ai_weight = {
			weight = 1000

			# >- HARD BLOCK -<
			modifier = {
				factor = 1000
				OR = {
					has_global_flag = hga_ai_no_relic_trade
					has_country_flag = hga_ai_no_relic
				}
			}

			# >- OPINION -<
			modifier = {
				factor = 1.3
				opinion_level = { who = from level = excellent }
			}
			modifier = {
				factor = 1.1
				opinion_level = { who = from level = good }
			}
			modifier = {
				factor = 0.6
				opinion_level = { who = from level = poor }
			}
			modifier = {
				factor = 0.2
				opinion_level = { who = from level = terrible }
			}

			# >- OVERLORD / SUBJECT -<
			modifier = {
				factor = 0.3
				root = { is_overlord_to = from }
			}
			modifier = {
				factor = 1.4
				from = { is_overlord_to = root }
			}

			# >- RELATIVE FLEET POWER -<
			modifier = {
				factor = 0.6
				relative_power = { who = from category = fleet value = overwhelming }
			}
			modifier = {
				factor = 0.8
				relative_power = { who = from category = fleet value = superior }
			}
			modifier = {
				factor = 1.15
				relative_power = { who = from category = fleet value = inferior }
			}
			modifier = {
				factor = 1.3
				relative_power = { who = from category = fleet value = pathetic }
			}

			# >- PER-RELIC CUSTOM MODIFIERS -<
			$CUSTOM_AI$
		}
	}
# /
