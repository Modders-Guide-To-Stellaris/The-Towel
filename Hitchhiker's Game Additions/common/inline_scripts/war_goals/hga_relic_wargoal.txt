# Usage:
#inline_script = {
#	script = war_goals/hga_relic_wargoal
#	WG_KEY     = hga_wg_seize_warlock
#	CB_KEY     = hga_cb_seize_warlock
#	RELIC_ID   = r_unbidden_warlock
#	FLAG_KEY   = hga_cannot_trade_warlock
#	CUSTOM_AI  = ""
#}
#
# CUSTOM_AI accepts raw modifier { } blocks to inject per-relic weighting.
# Leave as "" for default logic only.

# War goal definition
	$WG_KEY$ = {
		casus_belli = $CB_KEY$
		set_defender_wargoal = wg_humiliation

		should_ignore_federations = yes
		hide_if_no_cb = yes
		cede_claims = no

		allowed_peace_offers = {
			demand_surrender
			status_quo
			surrender
		}

		possible = {
			from = { has_relic = $RELIC_ID$ }
			NOT = { has_global_flag = hga_no_relic_war }
		}

		on_wargoal_set = {
			set_country_flag = $FLAG_KEY$
		}

		on_status_quo = {
			remove_country_flag = $FLAG_KEY$
		}

		on_accept = {
			remove_country_flag = $FLAG_KEY$
			if = {
				limit = { from = { has_relic = $RELIC_ID$ } }
				from = { remove_relic = $RELIC_ID$ }
				add_relic = $RELIC_ID$
			}
		}

		ai_weight = {
			# Base desire to go to war for a relic
			weight = 100

			# >- HARD BLOCK -<
			modifier = {
				factor = 0
				has_global_flag = hga_ai_no_relic_war
			}

			# >- NEVER DECLARE WHEN HOPELESSLY OUTMATCHED -<
			modifier = {
				factor = 0
				relative_power = { who = from category = fleet value = pathetic }
			}
			modifier = {
				factor = 0.1
				relative_power = { who = from category = fleet value = inferior }
			}

			# >- PREFER TO STRIKE WHEN POWERFUL -<
			modifier = {
				factor = 1.5
				relative_power = { who = from category = fleet value = superior }
			}
			modifier = {
				factor = 2.0
				relative_power = { who = from category = fleet value = overwhelming }
			}

			# >- OPINION: Don't war friends — trade instead -<
			modifier = {
				factor = 0
				opinion_level = { who = from level = excellent }
			}
			modifier = {
				factor = 0.2
				opinion_level = { who = from level = good }
			}
			modifier = {
				factor = 1.5
				opinion_level = { who = from level = poor }
			}
			modifier = {
				factor = 2.0
				opinion_level = { who = from level = terrible }
			}

			# >- ETHICS -<
			modifier = {
				factor = 1.5
				has_ethic = ethic_fanatic_militarist
			}
			modifier = {
				factor = 1.25
				has_ethic = ethic_militarist
			}
			modifier = {
				factor = 0.3
				has_ethic = ethic_fanatic_pacifist
			}
			modifier = {
				factor = 0.6
				has_ethic = ethic_pacifist
			}

			# >- OVERLORD/SUBJECT -<
			# Subjects should not start wars independently
			modifier = {
				factor = 0.1
				is_subject = yes
			}
			# Overlords are more emboldened
			modifier = {
				factor = 1.3
				is_overlord_to = from
			}

			# >- PER-RELIC CUSTOM MODIFIERS -<
			$CUSTOM_AI$
		}
	}
# /
