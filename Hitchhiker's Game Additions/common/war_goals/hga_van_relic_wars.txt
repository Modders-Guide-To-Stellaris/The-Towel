hga_wg_seize_warlock = {
    casus_belli = hga_cb_seize_warlock
    set_defender_wargoal = wg_humiliation

    should_ignore_federations = yes
    hide_if_no_cb = yes
    cede_claims = no

    allowed_peace_offers = {
        demand_surrender
        status_quo
        surrender
    }

    possible = {
        from = { has_relic = r_unbidden_warlock }
    }

    on_wargoal_set = {
        set_country_flag = hga_cannot_trade_warlock
    }

    on_status_quo = {
        remove_country_flag = hga_cannot_trade_warlock
    }

    on_accept = {
        remove_country_flag = hga_cannot_trade_warlock
        if = {
            limit = { from = { has_relic = r_unbidden_warlock } }
            from = { remove_relic = r_unbidden_warlock }
            add_relic = r_unbidden_warlock
        }
    }

    ai_weight = {
        # TODO
    }
}
