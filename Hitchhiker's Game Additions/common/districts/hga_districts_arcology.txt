hga_district_food_arcology = {
    inline_script = {
        script = districts/hga_arcology_districts
        ICON              = district_farming
        OVERLAY_ICON      = GFX_district_food
        RESOURCE_CATEGORY = planet_districts
        JOB_SCRIPT        = "jobs/farmers_add"
        JOB_AMOUNT        = @b3_jobs
        NAME_SCRIPT       = "districts/district_triggered_name_farming"
        FLAVOR_SCRIPT     = "districts/district_triggered_flavor_desc_farming"
        PREREQUISITE      = tech_industrial_farming
        HOUSING_INLINE    = "districts/hga_arcology_housing_food"
        CONVERT_TO = "
            district_farming
            district_rw_farming
            district_orchard_forests
            district_hive_3
        "
    }

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = { is_wilderness_empire = no }
        }
        modifier = {
            planet_housing_add = 200
        }
    }

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                has_valid_civic = civic_agrarian_idyll
            }
        }
        modifier = {
            planet_housing_add = 100 ## less because inline handles the other 300 :P
        }
    }

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                has_technology = tech_housing_agrarian_idyll
                has_valid_civic = civic_agrarian_idyll
            }
        }
        modifier = {
            planet_housing_add = 200
        }
    }

    destroy_trigger = {
        exists = owner
        AND = {
            owner = { is_ai = yes }
            owner = { country_uses_food = no }
        }
    }
}
